async function submitData() { const doctor = document.getElementById('doctorName').value; const patientId = document.getElementById('patientId').value.trim(); const patientName = document.getElementById('patientName').value.trim(); const visitId = document.getElementById('visitId').value.trim(); if (!doctor || !patientId || !patientName || !visitId) { showToast('Please fill all fields first', 'error'); return; } const entries = Object.entries(toothData) .filter(([,d]) => d.status || (d.treatments && d.treatments.length > 0)); if (!entries.length) { showToast('Chart at least one tooth first', 'error'); return; } const btn = document.getElementById('submitBtn'); btn.disabled = true; btn.textContent = 'Saving...'; const payload = { doctor, patientId, patientName, visitId, teeth: entries.map(([num, data]) => ({ num: String(num), desc: toothDescriptions[num] || '', status: data.status || '', treatments: (data.treatments || []).join('; '), done: (data.done || []).join('; '), notes: data.notes || '' })) }; try { const res = await fetch(SCRIPT_URL, { method: 'POST', headers: { 'Content-Type': 'text/plain' // 🔥 مهم جدًا مع Apps Script أحيانًا }, body: JSON.stringify(payload) }); const resultText = await res.text(); const result = JSON.parse(resultText); if (result.success) { showToast(`✓ Saved ${entries.length} teeth to Google Sheets!`); } else { showToast('Error: ' + (result.error || 'Unknown error'), 'error'); console.log(result); } } catch (err) { console.log(err); showToast('Network error saving data', 'error'); } btn.disabled = false; btn.textContent = '✓ Save to Google Sheets'; }
